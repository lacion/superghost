   - Add `onStepProgress?: OnStepProgress` to the config parameter of `executeAgent()`.
   - Inside `executeAgent()`, add `let stepCounter = 0;` before the `generateText()` call.
   - Add `experimental_onToolCallFinish` option to the `generateText()` call:
     ```
     experimental_onToolCallFinish: config.onStepProgress
       ? (event: any) => {
           if (event.success) {
             stepCounter++;
             const input = (event.toolCall.input ?? {}) as Record<string, unknown>;
             config.onStepProgress!({
               stepNumber: stepCounter,
               toolName: event.toolCall.toolName,
               input,
               description: describeToolCall(event.toolCall.toolName, input),
             });
           }
         }
       : undefined,
     ```
